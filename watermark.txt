        <!-- Watermark Options -->

                <option value="Times New Roman">Times New Roman</option>

                <option value="top-left">Top Left</option>

    <!-- Include JSZip library -->
    <script src="https://cdn.jsdelivr.net/npm/jszip@3.7.1/dist/jszip.min.js"></script>

    <script>
        const imageInput = document.getElementById("imageInput");
        const watermarkTextInput = document.getElementById("watermarkText");
        const fontFamilyInput = document.getElementById("fontFamily");
        const fontSizeInput = document.getElementById("fontSize");
        const fontColorInput = document.getElementById("fontColor");
        const opacityInput = document.getElementById("opacity");
        const locationInput = document.getElementById("location");
        const quantityInput = document.getElementById("quantity");
        const canvas = document.getElementById("canvas");
        const ctx = canvas.getContext("2d");
        const downloadBtn = document.getElementById("downloadBtn");
        const thumbnailsContainer = document.getElementById("thumbnails");

        let selectedImages = [];
        let currentImageIndex = -1;

        // Update canvas with watermark applied
        function updateCanvas(image) {
            const watermarkText = watermarkTextInput.value;
            const fontFamily = fontFamilyInput.value;
            const fontSize = fontSizeInput.value;
            const fontColor = fontColorInput.value;
            const opacity = opacityInput.value;
            const location = locationInput.value;
            const quantity = quantityInput.value;

            // Clear the canvas and redraw the image
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.drawImage(image, 0, 0, canvas.width, canvas.height);

            ctx.globalAlpha = opacity;

            // Add watermark multiple times
            for (let i = 0; i < quantity; i++) {
                const x = getWatermarkPosition(location, "x", canvas.width, fontSize, watermarkText);
                const y = getWatermarkPosition(location, "y", canvas.height, fontSize, watermarkText);

                ctx.font = `${fontSize}px ${fontFamily}`;
                ctx.fillStyle = fontColor;
                ctx.fillText(watermarkText, x, y);
            }
        }

        // Determine watermark position based on the selected location
        function getWatermarkPosition(location, axis, canvasSize, fontSize, watermarkText) {
            const padding = 20;
            switch (location) {
                case "top-left":
                    return padding;
                case "top-right":
                    return canvasSize - padding - ctx.measureText(watermarkText).width;
                case "center":
                    return (canvasSize - ctx.measureText(watermarkText).width) / 2;
                default:
                    return padding;
            }
        }

        // Handle image input (including folder)
        imageInput.addEventListener("change", (e) => {
            const files = e.target.files;

            selectedImages = [];
            currentImageIndex = -1;
            thumbnailsContainer.innerHTML = ""; // Clear existing thumbnails

            // Read each selected image
            Array.from(files).forEach((file) => {
                // Check if the file is an image
                if (file.type.startsWith("image/")) {
                    const reader = new FileReader();
                    reader.onload = function (event) {
                        const img = new Image();
                        img.onload = function () {
                            selectedImages.push(img);
                            
                            // Create a thumbnail for the image
                            const thumbnail = document.createElement("img");
                            thumbnail.src = event.target.result;
                            thumbnail.addEventListener("click", () => {
                                currentImageIndex = selectedImages.indexOf(img);
                                canvas.width = img.width;
                                canvas.height = img.height;
                                updateCanvas(img);
                            });
                            thumbnailsContainer.appendChild(thumbnail);
                        };
                        img.src = event.target.result;
                    };
                    reader.readAsDataURL(file);
                }
            });
        });

        // Update the canvas whenever user changes options
        [watermarkTextInput, fontFamilyInput, fontSizeInput, fontColorInput, opacityInput, locationInput, quantityInput].forEach(input => {
            input.addEventListener("input", () => {
                if (currentImageIndex !== -1) {
                    updateCanvas(selectedImages[currentImageIndex]);
                }
            });
        });

        // Download all watermarked images as a ZIP
        downloadBtn.addEventListener("click", async () => {
            const zip = new JSZip();

            for (let i = 0; i < selectedImages.length; i++) {
                const image = selectedImages[i];

                // Create a canvas to render the watermarked image
                const tempCanvas = document.createElement("canvas");
                const tempCtx = tempCanvas.getContext("2d");
                tempCanvas.width = image.width;
                tempCanvas.height = image.height;

                // Apply the watermark
                updateCanvas(image);

                const imageDataUrl = tempCanvas.toDataURL("image/png");
                const imageBlob = await fetch(imageDataUrl).then(res => res.blob());

                // Add the image to the ZIP file
                zip.file(`watermarked-image-${i + 1}.png`, imageBlob);
            }

            // Generate and download the ZIP file
            zip.generateAsync({ type: "blob" }).then(function (content) {
                const a = document.createElement("a");
                a.href = URL.createObjectURL(content);
                a.download = "watermarked-images.zip";
                a.click();
            });
        });
    </script>
